function normalizeText(text) { return text .replace(/@/g, 'o') .replace(/1/g, 'i') .replace("0","o"); } function swearFilter(text, swears, placeholder = '*') { text = normalizeText(text.toLowerCase()); let normalizedSwears = swears.map(normalizeText); for (let i = 0; i < normalizedSwears.length; i++) { let escapedSwear = normalizedSwears[i].replace(/[-\/\\^$*+?.()|[\]{}]/g, '\\$&'); let pattern = new RegExp("\\b" + escapedSwear.split('').join('[^a-zA-Z0-9]*') + "\\b", "gi"); let placeholderStr = placeholder.repeat(normalizedSwears[i].length); text = text.replace(pattern, placeholderStr); } return text; } // Example usage: let text = "This is a sentence with some bad words like i d i o t and m@r0n."; let swears = ["idiot", "moron", "badword"]; alert(swearFilter(text, swears)); /* NOT MY API */
